% The skeleton for producing my publication list in sections Uses biblatex and
% biber to customized the bibliography output. Sections defined by a
% combination of keyword= and custom filtering. See
% http://www.ctan.org/pkg/biblatex

\documentclass[letterpaper,11pt]{article}
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{mathpazo}

\usepackage[defernumbers=true, bibstyle=authoryear, backend=biber, maxbibnames=10,
    url=false, sorting=ydnt]{biblatex}

\usepackage{hyperref}
\hypersetup{ % turn off the boxes round URLs
    colorlinks,
    citecolor=black,
    filecolor=black,
    linkcolor=black,
    urlcolor=black
}

\addbibresource{matthew_brett.bib}
\defbibfilter{methodsPlus}{%
    keyword=computing or keyword=methods
}
\AtEveryBibitem{\clearlist{language}} % clears language

\begin{document}

\nocite{*}

\printbibheading[title={Matthew Brett -- publications}]

\printbibliography[heading=subbibliography,
    filter=methodsPlus,
    keyword=article,
    notkeyword=omit,
    title={Imaging methods: articles and book chapters}]
\printbibliography[heading=subbibliography,
    filter=methodsPlus,
    keyword=abstract,
    notkeyword=omit,
    title={Imaging methods: abstracts}]
\printbibliography[heading=subbibliography,
    keyword=movethink,
    keyword=article,
    notkeyword=omit,
    title={Imaging of movement and thinking: articles}]
\printbibliography[heading=subbibliography,
    keyword=movethink,
    keyword=abstract,
    notkeyword=omit,
    title={Imaging of movement and thinking: abstracts}]
\printbibliography[heading=subbibliography,
    keyword=others,
    notkeyword=omit,
    title={About other things}]

\printbibheading[title={Selected software}]

\printbibliography[heading=subbibliography,
    keyword=software,
    keyword=maintainer,
    notkeyword=omit,
    title={Software - lead developer}]
\printbibliography[heading=subbibliography,
    keyword=software,
    keyword=developer,
    notkeyword=omit,
    title={Software - core developer}]

\printbibheading[title={Selected online tutorials}]

\printbibliography[heading=subbibliography,
    heading=none,
    keyword=online,
    notkeyword=omit]

\end{document}
